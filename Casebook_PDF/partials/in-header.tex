\titleformat{\chapter}
  [block]
  {\raggedright\sffamily\huge\color{Maroon}}
  {\thechapter ~~}
  {0em}
  {}
  []

\titleformat{\section}
  [block]
  {\needspace{12\baselineskip}\raggedright\rmfamily\LARGE\color{Maroon}}
  {\thesection ~~}
  {0em}
  {}
  []

% Case, statute, rule, etc. title
\titleformat{\subsection}
  [block]
  {\needspace{6\baselineskip}\raggedright\rmfamily\Large\color{Maroon}}
  {\thesubsection ~~}
  {0em}
  {}
  [\vspace{-1.3em}\textcolor{Maroon}{\rule{\textwidth}{1pt}}]

% Case, etc. first heading
\titleformat{\subsubsection}
  [block]
  {\needspace{3\baselineskip}\raggedright\rmfamily\large\bfseries}
  {\thesubsubsection ~~}
  {0em}
  {}
  []

% Case, etc. second heading
\titleformat{\paragraph}
  [block]
  {\needspace{3\baselineskip}\raggedright\rmfamily\large\itshape}
  {\theparagraph}
  {0em}
  {}
  []
